# COMPLETE SETUP COMMANDS - COPY AND PASTE EVERYTHING BELOW

# 1. CREATE AND ACTIVATE VIRTUAL ENVIRONMENT
python -m venv venv
# On Windows:
venv\Scripts\activate
# On Mac/Linux:
# source venv/bin/activate

# 2. INSTALL BACKEND DEPENDENCIES
pip install fastapi
pip install uvicorn
pip install python-multipart
pip install -U google-genai
pip install firebase-admin

# 3. INSTALL FRONTEND DEPENDENCIES (in new terminal)
cd linaw-app
npm install axios
npm install uuid
npm install --save-dev @types/uuid

# 4. START BACKEND SERVER (in first terminal, venv activated)
uvicorn main:app --reload

# 5. START FRONTEND SERVER (in second terminal)
cd linaw-app
npm run dev
